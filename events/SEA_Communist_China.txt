country_event = {
	id = SEA_PRC_request_aid.5 
	title = SEA_PRC_request_aid.5.t
	desc = SEA_PRC_request_aid.5.desc
	picture = GFX_report_event_soviet_tanks 
	
	is_triggered_only = yes

	fire_only_once = no

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					has_dlc = "No Step Back"
				}
				set_technology = { improved_medium_tank_chassis = 1 popup = no } ## Tech pop up no
				create_equipment_variant = {
					name = "T-34"
					type = medium_tank_chassis_2
					parent_version = 0
					modules = {
						main_armament_slot = tank_medium_cannon
						turret_type_slot = tank_medium_two_man_tank_turret
						suspension_type_slot = tank_christie_suspension
						armor_type_slot = tank_welded_armor
						engine_type_slot = tank_diesel_engine
						special_type_slot_1 = sloped_armor
					}
					upgrades = {
						tank_nsb_engine_upgrade = 10
						tank_nsb_armor_upgrade = 7
					}
					icon = "GFX_SOV_improved_medium_tank_medium"
					allow_without_tech = yes
				}
			}
			else = {
				set_technology = { improved_medium_tank = 1 popup = no }
				create_equipment_variant = { ## Create non-NSB T34 variant and tech 
					name = "T-34"
					type = medium_tank_equipment_2 
					icon = "GFX_SOV_improved_medium_tank_medium"
				}
			}
		}
	}

	option = {
		name = SEA_PRC_request_aid.5.a

		if = {
			limit = {
				has_dlc = "No Step Back"
			}
			
			custom_effect_tooltip = PRC_appeal_for_foreign_aid_all_tanks_tt

			hidden_effect = {
				add_equipment_to_stockpile = {  
					type = medium_tank_chassis_2
					amount = 250
					#producer = SOV ## Can't have SOV as producer if they don't have that exact variant name, so more reliable this way 
					variant_name = "T-34" 
				}
			}
		}
		else = {
			add_equipment_to_stockpile = { 
				type = medium_tank_equipment_2
				amount = 250
				#producer = SOV
				variant_name = "T-34"
			}
		}
	
		
		FROM = { ## Remove from SOV

			if = {
				limit = {
					has_dlc = "No Step Back"
				}
				add_equipment_to_stockpile = { 
					type = medium_tank_chassis_2
					amount = -250
					producer = THIS
					#variant_name = "T-34/85"
				}
			}
			else = {
				add_equipment_to_stockpile = { 
					type = medium_tank_equipment_2
					amount = -250
					producer = THIS
					#variant_name = "T-34/85"
				}
			}			
		}
	}

}

## Guotao leaving for KMT
country_event = { 
	id = SEA_PRC_zhang_guotao_events.4
	title = SEA_PRC_zhang_guotao_events.4.t
	desc = SEA_PRC_zhang_guotao_events.4.desc
	picture = GFX_report_event_PRC_guotao_leaving


#	is_triggered_only = yes
	
	trigger = {
		original_tag = PRC
		date > 1938.5.14
		has_dlc = "No Compromise, No Surrender"
		NOT = {
			has_completed_focus = PRC_the_unwilling_diarchy_focus
			focus_progress = {
				focus = PRC_the_unwilling_diarchy_focus
				progress > 0
			}
		}
		NOT = {
			PRC_zhang_guotao = {
				has_character_flag = PRC_dismissed_advisor
				has_character_flag = PRC_zhang_guotao_left
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	fire_only_once = yes


	option = { ### Traitor to our cause
		name = SEA_PRC_zhang_guotao_events.4.a

		ai_chance = {
			base = 50

		}

		custom_effect_tooltip = PRC_zhang_guotao_left_tt

		set_country_flag = PRC_zhang_guotao_left

		CHI = {
			country_event = {
				id = SEA_PRC_zhang_guotao_events.5
				hours = 2
			}
		}

	}

	option = { ### Let him stay
		name = SEA_PRC_zhang_guotao_events.4.b

		add_political_power = -25
		
		trigger = {
			PRC_zhang_guotao = {
				NOT = {
					has_character_flag = PRC_dismissed_advisor ## This option is not visible if the player exiles Guotao earlier than this event triggers
				}
			}
		}

		ai_chance = {
			base = 5

		}

	}

}

				array = global.PRC_generals_flipped
			}
		}
	}
}

	title = SEA_PRC_political_events.20.t
	desc = SEA_PRC_political_events.20.desc
	picture = GFX_report_event_PRC_generals_flock

	is_triggered_only = yes 


	option = { ### We have received new generals
		name = SEA_PRC_political_events.20.a

		every_character = {
			limit = {
				is_in_array = {
					array = global.PRC_generals_flipped
					value = THIS.id
				}
				is_unit_leader = yes
			}
			gain_xp = 50 ## Mainly to show all the new generals in tooltip
		}
		add_manpower = 5000 ## Representing their closest forces
	}
}
