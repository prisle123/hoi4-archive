
# FOREIGN POLICY
foreign_politics = {
	
	PSR_core_state = {
		icon = GFX_decision_infiltrate_state
	
		allowed = {
			original_tag = PSR
		}
		
		visible = {
			is_ally_with = JAP
		}
		
		available = {
			NOT = { has_country_flag = PSR_coring_state_in_progress }
		}

		state_target = any_controlled_state
	    
	    target_trigger = {
	        FROM = {
	        	NOT = { is_core_of = ROOT }
	        	is_fully_controlled_by = ROOT
	        	is_core_of = PHI
	        }
	    }
	    on_map_mode = map_and_decisions_view 

	    cost = 150		

		fire_only_once = yes

		days_remove = 84

		ai_will_do = {
			base = 100
		}

		complete_effect = {
			hidden_effect = {
				set_country_flag = { flag = PSR_coring_state_in_progress value = 1 days = 31 }
			}
		}

		remove_effect = {
			hidden_effect = {
				clr_country_flag = PSR_coring_state_in_progress
			}
			IF = {
				limit = {
					FROM = {
						NOT = {
							is_owned_by = ROOT
						}
					}
				}
				FROM = {
					add_core_of = ROOT
					transfer_state_to = ROOT
				}
			}
			ELSE = {
				FROM = {
					add_core_of = ROOT
				}
			}
		}

		cancel_trigger = {
			FROM = {
				NOT = { is_fully_controlled_by = ROOT }
			}
		}

		cancel_effect = {
			hidden_effect = {
				clr_country_flag = PSR_coring_state_in_progress
			}
		}
	}


}
